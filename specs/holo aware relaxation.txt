import os
import json
import pymol
from pymol import cmd
import subprocess

def fix_pdb_elements(pdb_file):
    """Restores element symbols in columns 77-78 for RDKit/Meeko."""
    with open(pdb_file, 'r') as f:
        lines = f.readlines()
    fixed_lines = []
    for line in lines:
        if line.startswith("ATOM") or line.startswith("HETATM"):
            atom_name = line[12:16].strip()
            element = atom_name[0] if not atom_name[0].isdigit() else atom_name[1]
            line = line.rstrip().ljust(76)
            line = line[:76] + element.rjust(2) + "\n"
        fixed_lines.append(line)
    with open(pdb_file, 'w') as f:
        f.writelines(fixed_lines)

def process_and_convert(pdb_path, meta_path, out_dir):
    if not os.path.exists(pdb_path) or not os.path.exists(meta_path):
        return

    with open(meta_path, 'r') as f:
        meta = json.load(f)

    cmd.reinitialize()
    cmd.load(pdb_path, "receptor")

    # 1. IDENTIFY BZD CHAINS
    bzd_chains = [ch for ch, name in meta.get("chains", {}).items() if "_bzd" in name]
    if not bzd_chains: return
    chain_sel = " or ".join([f"chain {c}" for c in bzd_chains])
    
    # 2. PRE-PDB2PQR STRIP: Keep Protein + Ligands (Ignore Waters/Ions)
    # This ensures the ligand stays to 'block' side-chains from moving into the pocket
    cmd.remove(f"not ({chain_sel} or hetatm)")
    cmd.remove("resn HOH+WAT+NA+CL") 

    base_name = os.path.basename(pdb_path).split('.')[0]
    raw_holo_pdb = os.path.join(out_dir, f"{base_name}_holo_raw.pdb")
    cmd.save(raw_holo_pdb, "receptor")

    # 3. PROTONATE (HOLO): No --nodebump to allow side-chain relaxation around ligand
    protonated_holo_pdb = os.path.join(out_dir, f"{base_name}_holo_prot.pdb")
    pdb2pqr_cmd = [
        "pdb2pqr", "--ff=PARSE", "--titration-state-method=propka",
        "--with-ph=7.4", "--keep-chain", raw_holo_pdb, protonated_holo_pdb
    ]
    subprocess.run(pdb2pqr_cmd, check=True, capture_output=True)

    # 4. FINAL STRIP: Remove the ligand to get the clean receptor
    cmd.reinitialize()
    cmd.load(protonated_holo_pdb, "prot_receptor")
    cmd.remove("not polymer.protein") # Now we nuke the ligand
    
    final_receptor_pdb = os.path.join(out_dir, f"{base_name}_final_receptor.pdb")
    cmd.save(final_receptor_pdb, "prot_receptor")

    # 5. SANITIZE & CONVERT
    fix_pdb_elements(final_receptor_pdb)
    
    out_basename = os.path.join(out_dir, base_name)
    meeko_cmd = [
        "mk_prepare_receptor.py",
        "-i", final_receptor_pdb,
        "-o", out_basename,
        "-p"
    ]
    subprocess.run(meeko_cmd, check=True, capture_output=True)
    print(f"[+] Created 0,0,0 Aligned Receptor: {out_basename}.pdbqt")

    cmd.delete("all")

Summary of Rules for the Pipeline:

    Alignment: Every step preserves the original XYZ matrix. Your 0,0,0 pocket remains valid.

    Holo-Structure Integrity: By keeping the ligand during pdb2pqr, the protein side-chains are protonated in their "active" (opened) conformation.

    No Nodebump: By omitting --nodebump, the script is allowed to rotate side-chain hydrogens and chi-angles to resolve any impossible clashes created by the cryo-EM model's resolution limits.

    Sanitization: The fix_pdb_elements function guarantees Meeko will not crash on the PDB2PQR output.

The flag is --nodebump.

Adding this flag turns off the "de-bumping" (atom-shifting) behavior.

    Without the flag: PDB2PQR will automatically "bump" (rotate or shift) side chains to resolve any atomic overlaps it detects.

    With the flag (--nodebump): PDB2PQR is forbidden from moving heavy atoms. It will place hydrogens, but it will not "fix" the protein's conformation to avoid clashes.

Final Code Implementation

Since you are dealing with receptors that have ligands bound (Holo-structures), you should omit the --nodebump flag. This allows the protein side chains to relax and shift slightly to accommodate the ligand's presence, ensuring the binding pocket remains "open" and chemically valid for those specific coordinates.
